FDCPE_Q0: FDCPE port map (Q(0),Q_D(0),C,Q_CLR(0),Q_PRE(0));
     Q_D(0) <= ((ALOAD AND D(0))
      OR (NOT ALOAD AND NOT tmp(0).LFBK));
     Q_CLR(0) <= (ALOAD AND NOT D(0));
     Q_PRE(0) <= (ALOAD AND D(0));
FDCPE_Q1: FDCPE port map (Q(1),Q_D(1),C,Q_CLR(1),Q_PRE(1));
     Q_D(1) <= ((ALOAD AND D(1))
      OR (NOT ALOAD AND tmp(1).LFBK AND NOT Q(0).PIN)
      OR (NOT ALOAD AND NOT tmp(1).LFBK AND Q(0).PIN));
     Q_CLR(1) <= (ALOAD AND NOT D(1));
     Q_PRE(1) <= (ALOAD AND D(1));
FTCPE_Q2: FTCPE port map (Q(2),Q_T(2),C,Q_CLR(2),Q_PRE(2));
     Q_T(2) <= ((ALOAD AND D(2) AND NOT tmp(2).LFBK)
      OR (ALOAD AND NOT D(2) AND tmp(2).LFBK)
      OR (NOT ALOAD AND Q(0).PIN AND Q(1).PIN));
     Q_CLR(2) <= (ALOAD AND NOT D(2));
     Q_PRE(2) <= (ALOAD AND D(2));
FTCPE_Q3: FTCPE port map (Q(3),Q_T(3),C,Q_CLR(3),Q_PRE(3));
     Q_T(3) <= ((ALOAD AND D(3) AND NOT tmp(3).LFBK)
      OR (ALOAD AND NOT D(3) AND tmp(3).LFBK)
      OR (NOT ALOAD AND Q(0).PIN AND Q(1).PIN AND Q(2).PIN));
     Q_CLR(3) <= (ALOAD AND NOT D(3));
     Q_PRE(3) <= (ALOAD AND D(3));
